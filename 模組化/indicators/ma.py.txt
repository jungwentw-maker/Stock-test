import pandas as pd
from typing import Iterable

def add_ma(df: pd.DataFrame, windows: Iterable[int] = (5, 20, 60, 120),
           close_col: str = "Close", prefix: str = "MA") -> pd.DataFrame:
    """
    添加多條移動平均線。
    """
    df = df.copy()
    for w in windows:
        df[f"{prefix}{w}"] = df[close_col].rolling(w).mean()
    return df
